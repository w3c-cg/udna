# Verify category hash
import hashlib

def verify_category_hash(category_file):
    with open(category_file, 'r') as f:
        data = json.load(f)
    
    # Recompute hash
    json_str = json.dumps(data['vectors'], sort_keys=True, separators=(',', ':'))
    computed_hash = hashlib.blake2b(json_str.encode(), digest_size=32).hexdigest()
    
    return computed_hash == data['metadata']['hash']
